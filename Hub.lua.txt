local function findMyTycoon()
    local tycoons = workspace:FindFirstChild("Tycoons")
    if not tycoons then return nil end
    
    for _, tycoon in ipairs(tycoons:GetChildren()) do
        -- Skip any single loose items/purchases misidentified as a full plot
        if tycoon:IsA("Model") and #tycoon:GetChildren() > 3 then
            local ownerValue = tycoon:FindFirstChild("Owner") or tycoon:FindFirstChild("Player")
            if ownerValue then
                if ownerValue.Value == LP or ownerValue.Value == LP.Name or tostring(ownerValue.Value) == LP.Name then
                    return tycoon
                end
            end
            
            local tName = tycoon.Name:lower()
            if tName:find(LP.Name:lower()) or tName:find(LP.DisplayName:lower()) then
                return tycoon
            end
        end
    end
    return nil
end

local function instantTouch(part)
    local char = LP.Character
    local root = char and (char:FindFirstChild("HumanoidRootPart") or char:FindFirstChildWhichIsA("BasePart"))
    if root and part and part:IsA("BasePart") then
        firetouchinterest(root, part, 0)
        firetouchinterest(root, part, 1)
    end
end

local padsByBase, activeLoops = {}, {}
local allowedBases = {Stone=true, Magic=true, Storm=true, Robotic=true, Blue=true}
local excludedBases = {Insanity=true, Giant=true, Dark=true, Spike=true, Web=true, Strong=true}
local toolToBase = {["Energy Sword"]="Stone", ["Staff"]="Magic", ["Axe"]="Storm", ["Fist"]="Robotic", ["Blue Sword"]="Blue"}
local PAD_RANGE = 1000
local TOUCH_BURST = 8

local function registerPad(pad)
    local base = pad.Parent and pad.Parent.Parent
    if not base or excludedBases[base.Name] or not allowedBases[base.Name] then return end
    padsByBase[base.Name] = padsByBase[base.Name] or {}
    table.insert(padsByBase[base.Name], pad)
end

local function hasSpecificTool(toolName)
    local function scan(container)
        if not container then return false end
        for _, t in ipairs(container:GetChildren()) do if t:IsA("Tool") and (t.Name == toolName or t.Name:lower():find("blue")) then return true end end
        return false
    end
    return scan(LP.Backpack) or scan(LP.Character)
end

local function getClosestPad(pads, root)
    local closest, dist
    for _, pad in ipairs(pads) do
        local d = (pad.Position - root.Position).Magnitude
        if d < PAD_RANGE and (not dist or d < dist) then
            dist = d
            closest = pad
        end
    end
    return closest
end

local function runParallelGrabLoop(toolName)
    if activeLoops[toolName] or not _G.Settings.ToolGrabber then return end
    local base = toolToBase[toolName]
    if not base then return end

    activeLoops[toolName] = true
    task.spawn(function()
        while activeLoops[toolName] and _G.Settings.ToolGrabber do
            if hasSpecificTool(toolName) then activeLoops[toolName] = nil break end
            local char = LP.Character
            local root = char and (char:FindFirstChild("HumanoidRootPart") or char:FindFirstChildWhichIsA("BasePart"))
            if not root then RunService.Heartbeat:Wait() continue end

            local pads = padsByBase[base]
            if pads then
                local pad = getClosestPad(pads, root)
                if pad then
                    for i = 1, TOUCH_BURST do
                        pcall(firetouchinterest, root, pad, 0)
                        pcall(firetouchinterest, root, pad, 1)
                    end
                end
            end
            RunService.Heartbeat:Wait()
        end
        activeLoops[toolName] = nil
    end)
end

local function triggerQuantumGrab()
    if not _G.Settings.ToolGrabber then return end
    for toolName in pairs(toolToBase) do runParallelGrabLoop(toolName) end
end

local function getToolPart(tool)
    if tool:FindFirstChild("Handle") and tool.Handle:IsA("BasePart") then return tool.Handle end
    if tool.PrimaryPart and tool.PrimaryPart:IsA("BasePart") then return tool.PrimaryPart end
    for _, v in ipairs(tool:GetDescendants()) do if v:IsA("BasePart") then return v end end return nil
end

local function updateToolCache()
    table.clear(cachedToolParts)
    if not LP.Character then return end
    for _, tool in ipairs(LP.Character:GetChildren()) do
        if tool:IsA("Tool") then local part = getToolPart(tool) if part then table.insert(cachedToolParts, part) end end
    end
    for _, tool in ipairs(LP.Backpack:GetChildren()) do
        if tool:IsA("Tool") then local part = getToolPart(tool) if part then table.insert(cachedToolParts, part) end end
    end
end

local function fixToolPhysics(tool)
    if tool:IsA("Tool") then
        tool.GripPos = Vector3.new(0, 0, 0)
        local part = getToolPart(tool)
        if part then 
            part.CanCollide = false 
            part.Massless = true 
            local bv = part:FindFirstChildOfClass("BodyVelocity")
            if not bv then
                bv = Instance.new("BodyVelocity")
                bv.MaxForce = Vector3.new(400000, 400000, 400000)
                bv.Velocity = Vector3.new(0, 0, 0)
                bv.Parent = part
            end
        end
    end
end

local function equipTools()
    local char, bp = LP.Character, LP.Backpack
    if not char or not bp then return end
    for _, t in ipairs(bp:GetChildren()) do 
        if t:IsA("Tool") and t.Parent ~= char then 
            t.Parent = char 
            fixToolPhysics(t)
        end 
    end
end

local function activateTools()
    local char = LP.Character if not char then return end
    for _, t in ipairs(char:GetChildren()) do 
        if t:IsA("Tool") then 
            pcall(t.Activate, t) 
        end 
    end
end

local SpawnDeltWith = false
local function runInstantSpawnSetup(char)
    if not char then return end
    SpawnDeltWith = true
    pcall(function()
        char:WaitForChild("HumanoidRootPart", 5)
        hookCharacterHumanoid(char)
        updateToolCache()
        
        if _G.Settings.GodMode then
            applyStableGodMode(char)
        end
        
        for _, t in ipairs(char:GetChildren()) do if t:IsA("Tool") then fixToolPhysics(t) end end
        char.ChildAdded:Connect(function(c) if c:IsA("Tool") then task.wait() updateToolCache() fixToolPhysics(c) end end)
        if _G.Settings.ToolGrabber then triggerQuantumGrab() end
    end)
end

RunService.Heartbeat:Connect(function(dt)
    local char = LP.Character
    local myRoot = char and (char:FindFirstChild("HumanoidRootPart") or char:FindFirstChildWhichIsA("BasePart"))
    if not char then return end

    if _G.Settings.UseTools then 
        equipTools()
        activateTools()
    end

    if _G.Settings.GodMode then
        pcall(function()
            for _, child in ipairs(char:GetChildren()) do
                if child:IsA("BasePart") and child.Name ~= "HumanoidRootPart" then
                    child.CanTouch = false
                    child.CanCollide = false
                end
            end
            local hum = char:FindFirstChildOfClass("Humanoid")
            if hum and hum.Health > 0 and hum.Health < hum.MaxHealth then
                hum.Health = hum.MaxHealth
            end
        end)
    end

    if _G.Settings.HitboxExpander then
        for _, p in ipairs(Players:GetPlayers()) do
            if p ~= LP and p.Character then
                local hrp = p.Character:FindFirstChild("HumanoidRootPart")
                if hrp then
                    pcall(function()
                        hrp.Size = Vector3.new(_G.Settings.HitboxSize, _G.Settings.HitboxSize, _G.Settings.HitboxSize)
                        hrp.Transparency = 0.85
                        hrp.CanCollide = false
                    end)
                end
            end
        end
    end

    if _G.Settings.Loopbring and myRoot then
        for name in pairs(_G.BringTargets) do
            local p = Players:FindFirstChild(name)
            local tRoot = p and p.Character and p.Character:FindFirstChild("HumanoidRootPart")
            if tRoot then
                local targetPosition = myRoot.Position + (myRoot.CFrame.LookVector * 2)
                tRoot.CFrame = CFrame.new(targetPosition, myRoot.Position) * CFrame.Angles(0, math.rad(180), 0)
            end
        end
    end

    if _G.Settings.KillAura and #getgenv().configs.TargetList > 0 then
        pcall(function()
            local auraReach = getgenv().configs.Size
            if _G.Settings.BlueSwordReach then
                auraReach = Vector3.new(auraReach.X * 1.5, auraReach.Y * 3.5, auraReach.Z * 1.5)
            end

            for _, targetPlayer in ipairs(getgenv().configs.TargetList) do
                if targetPlayer.Character and targetPlayer.Character:FindFirstChild("HumanoidRootPart") then
                    local tRoot = targetPlayer.Character.HumanoidRootPart
                    local distance = (myRoot.Position - tRoot.Position).Magnitude
                    
                    if distance <= auraReach.Y then
                        for _, tool in ipairs(char:GetChildren()) do
                            if tool:IsA("Tool") then
                                local touchPart = tool:FindFirstChildWhichIsA("TouchTransmitter", true)
                                touchPart = touchPart and touchPart.Parent or getToolPart(tool)
                                
                                if touchPart then
                                    firetouchinterest(touchPart, tRoot, 0)
                                    firetouchinterest(touchPart, tRoot, 1)

                                    if _G.Settings.DoubleDamage or _G.Settings.HitAmplifier then
                                        firetouchinterest(touchPart, tRoot, 0)
                                        firetouchinterest(touchPart, tRoot, 1)
                                    end
                                end
                            end
                        end
                    end
                end
            end
        end)
    end
end)
